   335→  const generateCouponCode = () => {
   336→    // 生成 8 位随机优惠码（大写字母+数字，去除易混淆字符）
   337→    const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789'
   338→    let code = ''
   339→    for (let i = 0; i < 8; i++) {
   340→      code += chars.charAt(Math.floor(Math.random() * chars.length))
   341→    }
   342→    setCouponForm(prev => ({ ...prev, code }))
   343→  }
   344→
   345→  const submitCoupon = async (e: React.FormEvent) => {
   346→    e.preventDefault()
   347→    setCouponError('')
   348→    if (!couponForm.code.trim()) {
   349→      setCouponError('请填写或生成优惠码')
   350→      return
   351→    }
   352→    if (couponForm.code.trim().length < 3) {
   353→      setCouponError('优惠码至少需要 3 个字符')
   354→      return
   355→    }
   356→    const dv = Number(couponForm.discountValue)
   357→    if (!Number.isFinite(dv) || dv <= 0) {
   358→      setCouponError('请输入有效的折扣值')
   359→      return
   360→    }
   361→    if (couponForm.discountType === 'percentage' && dv > 100) {
   362→      setCouponError('百分比折扣不能超过 100')
   363→      return
   364→    }
   365→    setCouponSubmitting(true)
   366→    try {
   367→      const payload: Record<string, unknown> = {
   368→        code: couponForm.code.trim(),
   369→        discountType: couponForm.discountType,
   370→        discountValue: dv,
   371→        planId: couponForm.planId,
   372→        maxUses: Number(couponForm.maxUses) > 0 ? Number(couponForm.maxUses) : 100,
   373→      }
   374→      if (couponForm.expiresAt) {
   375→        payload.expiresAt = new Date(couponForm.expiresAt).toISOString()
   376→      }
   377→      const res = await fetch('/api/admin/coupons', {
   378→        method: 'POST',
   379→        headers: { 'Content-Type': 'application/json' },
   380→        body: JSON.stringify(payload),
   381→      })
   382→      const data = await res.json()
   383→      if (res.ok) {
   384→        setCouponFormOpen(false)
   385→        setCouponForm({
   386→          code: '',
   387→          discountType: 'percentage',
   388→          discountValue: '',
   389→          planId: '',
   390→          maxUses: '100',
   391→          expiresAt: '',
   392→        })
   393→        await loadCoupons()
   394→      } else {
   395→        setCouponError(data.error || '创建失败')
   396→      }
   397→    } catch {
   398→      setCouponError('网络错误')
   399→    } finally {
   400→      setCouponSubmitting(false)
   401→    }
   402→  }

   546→  useEffect(() => {
